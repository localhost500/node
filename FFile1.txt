<body class="bg-light">

<div class="container mt-5 col-md-6">
    <div class="card p-3 shadow">

        <h4 class="text-center mb-3">Employee Registration</h4>

        <form onsubmit="return validateForm()">

            <input type="text" id="empNo" class="form-control mb-2" placeholder="Employee Number">
            <input type="text" id="empName" class="form-control mb-2" placeholder="Employee Name">
            <input type="number" id="salary" class="form-control mb-2" placeholder="Salary">

            <label>DOB</label>
            <input type="date" id="dob" class="form-control mb-2">

            <label>Joining Date</label>
            <input type="date" id="join" class="form-control mb-2">

            <button class="btn btn-success w-100 mt-2">Register</button>

        </form>

        <div id="msg" class="mt-3 text-center text-danger"></div>

    </div>
</div>

<script>
function validateForm() {
    let dob = new Date(document.getElementById("dob").value);
    let join = new Date(document.getElementById("join").value);
    let msg = document.getElementById("msg");

    msg.innerHTML = "";

    if (!dob || !join) {
        msg.innerHTML = "Select both dates";
        return false;
    }

    let age = join.getFullYear() - dob.getFullYear();

    if (age < 20) {
        msg.innerHTML = "Age must be at least 20 years";
        return false;
    }

    msg.className = "text-success";
    msg.innerHTML = "Employee Registered Successfully!";
    return false;
}
</script>

</body>
</html>

    <script>
        function validateLogin() {
            document.getElementById("userError").innerHTML = "";
            document.getElementById("emailError").innerHTML = "";
            document.getElementById("passError").innerHTML = "";
            document.getElementById("successMsg").style.display = "none";

            let username = document.getElementById("username").value.trim();
            let email = document.getElementById("email").value.trim();
            let password = document.getElementById("password").value;

            if (!/^[A-Za-z0-9]{4,}$/.test(username)) {
                document.getElementById("userError").innerHTML = "Min 4 chars";
                return false;
            }

            if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
                document.getElementById("emailError").innerHTML = "Invalid email";
                return false;
            }

            if (password.length < 6) {
                document.getElementById("passError").innerHTML = "Min 6 chars";
                return false;
            }

            document.getElementById("successMsg").style.display = "block";
            return false;
        }
    </script>
