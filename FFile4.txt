### run app.js ###

const express = require('express');
const bodyParser = require('body-parser');

const app = express();


app.use(bodyParser.urlencoded({ extended: false }));


app.get('/', (req, res) => {
    res.sendFile(__dirname + '/views/login.html');
});


app.post('/login', (req, res) => {
    const username = req.body.username;
    const password = req.body.password;

   
    if (username.length > 6) {
        res.send('Error: Username should not contain more than 6 characters');
    } else {
        res.send(
            'Login Successful <br>' +
            'Username: ' + username + '<br>' +
            'Password: ' + password
        );
    }
});

// Server
app.listen(5000, () => {
    console.log('Server running on http://localhost:5000');
});

### Run app.js ###

const express = require("express");
const methodOverride = require("method-override");
const app = express();

app.set("view engine", "ejs");

app.use(express.urlencoded({ extended: true }));
app.use(methodOverride("_method"));
app.use(express.static("public"));

let recipes = [
  { id: 1, name: "Pasta", description: "Delicious Italian pasta" },
  { id: 2, name: "Pizza", description: "Cheesy pizza with toppings" }
];

app.get("/", (req, res) => {
  res.render("home", { recipes });
});

app.get("/recipe/:id", (req, res) => {
  const recipe = recipes.find(r => r.id == req.params.id);
  res.render("recipe", { recipe });
});

app.post("/add", (req, res) => {
  const newRecipe = {
    id: Date.now(),
    name: req.body.name,
    description: req.body.description
  };
  recipes.push(newRecipe);
  res.redirect("/");
});

app.delete("/delete/:id", (req, res) => {
  recipes = recipes.filter(r => r.id != req.params.id);
  res.redirect("/");
});

app.listen(3001, () => {
  console.log("Server running on http://localhost:3001");
});

#### run app.js ###

const express = require('express');
const path = require('path');
const fs = require('fs');

const app = express();
const PORT = 3000;

// Serve static files from public folder
app.use(express.static(path.join(__dirname, 'public')));

// Route to display homepage
app.get('/', (req, res) => {
    res.sendFile(path.join(__dirname, 'public', 'index.html'));
});

// Download route
app.get('/download', (req, res) => {

    const filePath = path.join(__dirname, 'files', 'WF_Labbook.pdf');



    // Check if file exists
    fs.access(filePath, fs.constants.F_OK, (err) => {

        if (err) {
            return res.status(404).send("Error: File not found.");
        }

        // Force file download
        res.download(filePath, 'Downloaded_File.pdf', (err) => {

            if (err) {
                console.error("Download error:", err);
                res.status(500).send("Error while downloading file.");
            }

        });
    });
});

// Handle invalid routes
app.use((req, res) => {
    res.status(404).send("404 - Page Not Found");
});

// Start server
app.listen(PORT, '0.0.0.0', () => {
    console.log(`Server running at http://localhost:${PORT}`);
});

const express = require('express');
const bodyParser = require('body-parser');
const path = require('path');

const app = express();
const PORT = 4000;


app.set('view engine', 'ejs');
app.set('views', path.join(__dirname, 'views'));


app.use(bodyParser.urlencoded({ extended: true }));


app.get('/', (req, res) => {
    res.render('form');
});


app.post('/submit', (req, res) => {

    const employeeData = {
        name: req.body.name,
        id: req.body.id,
        department: req.body.department,
        salary: req.body.salary
    };

    res.render('result', { data: employeeData });
});


app.listen(PORT, () => {
    console.log(`Server running at http://localhost:${PORT}`);
});
