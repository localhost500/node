### Q3.page redirectionwhich contain at least two pages.

##Q3_loginapp/views.py

from django.shortcuts import render

def login_view(request):
    if request.method == "POST":
        username = request.POST.get('username')
        password = request.POST.get('password')

        if username == "admin" and password == "123":
            return render(request, 'Q3_loginapp/success.html')
        else:
            return render(request, 'Q3_loginapp/login.html', {'error': 'Invalid Credentials'})

    return render(request, 'Q3_loginapp/login.html')


##Q3_loginapp/urls.py

from django.urls import path
from . import views

urlpatterns = [
    path('', views.login_view),
]


## rkproject/urls.py
