#### Q1. Customer Collection insert 5 Records (MongoDB) ###

const { MongoClient } = require('mongodb');

const url = 'mongodb://127.0.0.1:27017';
const dbName = 'fymcs25';

async function run() {
    const client = new MongoClient(url);

    try {
        await client.connect();
        console.log('Connected to MongoDB');

        const db = client.db(dbName);
        const collection = db.collection('customers');

       
        const customers = await collection.find({}).toArray();
        console.log('All Customers:', customers);
       
        const deleteCondition = { customerId: 5 };
        const result = await collection.deleteOne(deleteCondition);

        console.log('Deleted Count:', result.deletedCount);

    } catch (err) {
        console.error('Error:', err);
    } finally {
        await client.close();
    }
}

run();

const { MongoClient } = require("mongodb");

const url = "mongodb://127.0.0.1:27017";
const dbName = "shopDB";

async function main() {
    try {
      
        const client = new MongoClient(url);
        await client.connect();
        console.log("Connected to MongoDB");

        const db = client.db(dbName);
        const collection = db.collection("customers");

        
        const customers = [
            { name: "Rupesh", city: "Pune", age: 22 },
            { name: "Rakesh", city: "Mumbai", age: 23 },
            { name: "Ritesh", city: "Nashik", age: 21 },
            { name: "Rakhav", city: "Nagpur", age: 24 },
            { name: "Rahul", city: "Kolhapur", age: 25 }
        ];

        await collection.insertMany(customers);
        console.log("5 customer records inserted");

        
        const result = await collection.find({}).toArray();

        console.log("Customer Records:");
        console.log(result);

        client.close();
    } catch (error) {
        console.log("Error:", error);
    }
}

main();

const { MongoClient } = require("mongodb");


const url = "mongodb://127.0.0.1:27017";
const dbName = "collegeDB";

async function run() {
    let client;

    try {
        
        client = new MongoClient(url);
        await client.connect();
        console.log("Database connected successfully");

        const db = client.db(dbName);
        const collection = db.collection("student");

       
        const students = [
            { roll: 1, name: "Rupesh", age: 22, course: "MSc" },
            { roll: 2, name: "Amit", age: 23, course: "MSc" },

            { roll: 4, name: "Priya", age: 24, course: "MSc" },
